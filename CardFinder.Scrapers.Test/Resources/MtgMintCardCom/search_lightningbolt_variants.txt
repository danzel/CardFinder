    function just_edition_submit()
    {
        var ed = $( "#ed" ).val();
        var ed_name = get_edition_eng_name(ed);
        if (  $.trim($('#keywords').val()) == ''  )
        {
           var  url = DomainNameBase + 'mtg/singles/' + ed_name + '/' + 'eng-reg' ;
           window.location = url;
           return false;  
        }
        else
        {
           return true;
        }
    }

    function go_edition_action_url(url,other_params,droupdown_name)
    {
      var e = document.getElementById(droupdown_name);
     // var edition_name = e.options[e.selectedIndex].text;
      var edition_id = e.options[e.selectedIndex].value;
      edition_name  =  get_edition_eng_name(edition_id);

      edition_name = edition_name.replace(/ /g,'-');
      edition_name = encodeURIComponent(edition_name);
      edition_name = edition_name.replace(/%26/g,'%2526');
      url = url + edition_name ;

      if ( other_params != '' )
      {
         url = url + '?' + other_params ;
      }

      //alert(url);
      location.href= url;

      document.mtg_search_change_editon.action = url;
    }

   function change_language_currency()
   {
      var interface_language = $("#language").val();
      var interface_currency = $("#currency").val();
      var request_uri = $("#request_uri").val();

      var request_uri_array = request_uri.split("?");
      var request_uri_params = '';
      var request_uri_page = '';

      if ( request_uri_array.length > 1 )
      {

          var request_uri_params = '';
          var request_uri_page = request_uri_array[0];
          var prmarr = request_uri_array[1].split("&");

          for ( var i = 0; i < prmarr.length; i++)
          {

             var tmparr = prmarr[i].split("=");

             if ( tmparr[0] != '' )
             {

                 if ( tmparr[0] == 'language' ||  tmparr[0] == 'currency' )
                 {
                     continue;
                 }
                 else
                 {
                     request_uri_params += tmparr[0] + '=' + tmparr[1] + '&';
                 }
             }
          }

           var redirect_uri = request_uri_page + '?' +  request_uri_params  + 'language=' + interface_language + '&currency=' + interface_currency;

      }
      else
      {
          var redirect_uri = request_uri + '?' + 'language=' + interface_language + '&currency=' + interface_currency;
      }

      window.location.replace(redirect_uri);
   }
